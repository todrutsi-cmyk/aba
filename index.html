function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const rows = sheet.getDataRange().getValues(); // 시트가 비어있으면 헤더 작성 if (rows.length === 1 && rows[0][0] === "") { sheet.appendRow(["Date", "SkillID", "SkillName", "Domain", "Mission", "Level", "Result", "Reinforcer", "Note", "Timestamp"]); return ContentService.createTextOutput(JSON.stringify({ status: "success", data: {} })) .setMimeType(ContentService.MimeType.JSON) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); } const data = {}; for (let i = 1; i < rows.length; i++) { const row = rows[i]; if (!row[0]) continue; let dateStr = row[0]; if (dateStr instanceof Date) { dateStr = Utilities.formatDate(dateStr, Session.getScriptTimeZone(), "yyyy-MM-dd"); } else { dateStr = String(dateStr).trim(); } const skillId = String(row[1]); if (!data[dateStr]) data[dateStr] = {}; data[dateStr][skillId] = { name: row[2], domain: row[3], mission: row[4], level: row[5] ? Number(row[5]) : null, result: row[6], reinforcer: row[7], note: row[8] }; } return ContentService.createTextOutput(JSON.stringify({ status: "success", data: data })) .setMimeType(ContentService.MimeType.JSON) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (!e || !e.postData || !e.postData.contents) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "No data" })) .setMimeType(ContentService.MimeType.JSON) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); } const postData = JSON.parse(e.postData.contents); const dateStr = String(postData.date).trim(); const entries = postData.entries || {}; const skillsList = postData.skills || []; const rows = sheet.getDataRange().getValues(); // 덮어쓰기: 같은 날짜 데이터 삭제 for (let i = rows.length - 1; i >= 1; i--) { let checkDate = rows[i][0]; if (checkDate instanceof Date) { checkDate = Utilities.formatDate(checkDate, Session.getScriptTimeZone(), "yyyy-MM-dd"); } else { checkDate = String(checkDate).trim(); } if (checkDate === dateStr) { sheet.deleteRow(i + 1); } } const timestamp = new Date(); Object.keys(entries).forEach(skillId => { const entry = entries[skillId]; if (!entry) return; const skillObj = skillsList.find(s => s.id === skillId) || {}; if (entry.result || entry.level || entry.mission || entry.note || entry.reinforcer) { sheet.appendRow([ dateStr, skillId, skillObj.name || entry.name || skillId, skillObj.domain || entry.domain || '', entry.mission || '', entry.level || '', entry.result || '', entry.reinforcer || '', entry.note || '', timestamp ]); } }); return ContentService.createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); } function doOptions(e) { return ContentService.createTextOutput("") .setMimeType(ContentService.MimeType.TEXT) .setHeaders({ 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', 'Access-Control-Max-Age': '86400' }); }
